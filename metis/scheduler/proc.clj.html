001  (ns metis.scheduler.proc
002    ^{:author "wactbprot"
003      :doc "Finds and starts the up comming tasks of a certain container."}
004    (:require [com.brunobonacci.mulog :as mu]
005              ))
006    
007  (defn filter-state
008    "Returns a vector of maps where `:value` is `s`."
009    [v s]
010    (filterv (fn [m] (= s (:value m))) v))
011  
012  (defn filter-seq-idx
013    "Returns a vector of maps where `:seq-idx` is `i`."
014    [v i]
015    (filterv (fn [m] (= i (:seq-idx m))) v))
016  
017  (defn all-error
018    "Returns all steps with the state `:error` for a given state vector
019    `v`"
020    [v]
021    (filter-state v "error"))
022  
023  (defn all-ready
024    "Returns all steps with the state `:ready` for a given state vector
025    `v`"
026    [v]
027    (filter-state v "ready"))
028  
029  (defn all-executed
030    "Returns all-executed entrys of the given vector `v`.
031  
032    Example:
033    ```clojure
034      (all-executed (stu/seq-idx->all-par
035                        [{:seq-idx 0 :par-idx 0 :state :executed}
036                         {:seq-idx 0 :par-idx 0 :state :ready}]
037                       0))
038    ;; =>
039    ;; [{:seq-idx 0 :par-idx 0 :state :executed}]
040    ```"
041    [v]
042    (filter-state v "executed"))
043  
044  (defn all-executed?
045    "Checks if all entries of map `m` are executed"
046    [v]
047    (= (count v) (count (all-executed v))))
048  
049  (defn errors?
050    "Checks if there are any errors in the
051    vector of maps `v`."
052    [v]
053    (not (empty? (all-error v))))
054  
055  (defn next-ready
056    "Returns a map (or `nil`) with the next step with state `:ready`."
057    [v]
058    (first (all-ready v)))
059  
060  (defn predecessors-executed?
061    "Checks if the steps of `v` before `i` are `all-executed?`. Returns
062    `true` for `i` equal to zero (or negative `i`)"
063    [v i]
064    (if (pos? i)
065      (every? true? (map
066                     (fn [j] (all-executed? (filter-seq-idx v j)))
067                     (range i)))
068        true))

074  (defn next-map
075    "The `next-map` function returns a map containing the next step to
076    start. See `cmp.state-test/next-map-i` for examples how `next-map`
077    should work.
078  
079    Example:
080    ```clojure
081     (next-map [{:seq-idx 0 :par-idx 0 :state :executed}
082                {:seq-idx 0 :par-idx 1 :state :executed}
083                {:seq-idx 1 :par-idx 0 :state :executed}
084                {:seq-idx 2 :par-idx 0 :state :executed}
085                {:seq-idx 3 :par-idx 0 :state :working}
086                {:seq-idx 3 :par-idx 1 :state :ready}])
087    ;; =>
088    ;; {:seq-idx 3 :par-idx 1 :state :ready}
089    ```"
090    [v]
091    (when-let [next-m (next-ready v)]
092      (when-let [i (:seq-idx next-m)]
093        (when (or (zero? i) (predecessors-executed? v i))
094          next-m))))
