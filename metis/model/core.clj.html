001  (ns metis.model.core
002    (:require [metis.config.interface :as c]
003              [metis.stmem.interface :as stmem]
004              [clojure.string :as string]
005              [metis.utils.interface :as utils]))
006  
007  ;;------------------------------
008  ;; exchange
009  ;;------------------------------
010  (defn build-exchange
011    "Builds the exchange interface."
012    [{mp-id :mp-id exch :Exchange}]  
013    (doseq [[k v] exch]
014      (stmem/set-val {:mp-id mp-id :struct :exch :exch (name k) :value v})))
015  
016  ;;------------------------------
017  ;; definition (cont & defins)
018  ;;------------------------------
019  (defn build-defin
020    "Builds the definition section."
021    [{defin :Definition :as m}]
022    (doall
023     (map-indexed
024      (fn [sdx s]
025        (doall
026         (map-indexed
027          (fn [pdx task]
028            (stmem/set-val (assoc m :func :defin :seq-idx sdx :par-idx pdx :value task))
029            (stmem/set-val (assoc m :func :state :seq-idx sdx :par-idx pdx :value "ready")))
030          s)))
031      defin)))
032  
033  ;;------------------------------
034  ;; container
035  ;;------------------------------
036  (defn build-container
037    "Builds a single container."
038    [{descr :Description title :Title ctrl :Ctrl elem :Element :as m}]
039    (stmem/set-val (assoc m :func :title :value title))
040    (stmem/set-val (assoc m :func :descr :value descr))
041    (stmem/set-val (assoc m :func :ctrl :value ctrl))
042    (stmem/set-val (assoc m :func :elem :value elem))
043    (build-defin m))
044  
045  (defn build-all-container
046    "Triggers the storing of the containers."
047    [{mp-id :mp-id conts :Container}]
048    (doall
049     (map-indexed
050      (fn [idx cont] (build-container (assoc cont  :mp-id mp-id :struct :cont :no-idx idx)))
051      conts)))
052  
053  ;;------------------------------
054  ;; definitions
055  ;;------------------------------
056  (defn build-conds
057    "Stores the definitions conditions."
058    [{cnds :Condition :as m}]
059    (doall
060     (map-indexed
061      (fn [jdx cnd]
062        (stmem/set-val (assoc m :seq-idx jdx :func :cond :value cnd)))
063      cnds)))
064  
065  (defn build-definitions
066    "Stores a definition given in the definition section
067    (second way beside container to provide definitions).  This includes
068    `DefinitionClass` and `Conditions`."
069    [{cls :DefinitionClass descr :ShortDescr conds :Condition :as m}]
070    (stmem/set-val (assoc m :func :descr :value descr))
071    (stmem/set-val (assoc m :func :cls :value cls))
072    (stmem/set-val (assoc m :func :ctrl :value "ready"))
073    (build-conds m)
074    (build-defin m))
075  
076  (defn build-all-definitions
077    "Triggers the storing of the definition section."
078    [{mp-id :mp-id defins :Definitions}]
079    (doall
080     (map-indexed
081      (fn [idx defin] (build-definitions (assoc defin :mp-id mp-id :struct :defins :no-idx idx)))
082      defins)))
083  
084  ;;------------------------------
085  ;; meta
086  ;;------------------------------
087  (defn build-meta
088    "Stores the meta data of an mpd:
089    
090    * standard
091    * name
092    * description
093    * number of containers
094    * number of definitions
095    "
096    [{mp-id :mp-id std :Standard name :Name descr :Description cont :Container defins :Definitions}]
097    (let [m {:mp-id mp-id :struct :meta}]
098      (stmem/set-val (assoc m :metapath :std :value std))
099      (stmem/set-val (assoc m :metapath :name :value name))
100      (stmem/set-val (assoc m :metapath :descr :value descr))
101      (stmem/set-val (assoc m :metapath :nd :value (count defins)))
102      (stmem/set-val (assoc m :metapath :nc :value (count cont)))))
103  
104  ;;------------------------------
105  ;; build mpd doc
106  ;;------------------------------
107  (defn build-mpd
108    [{mp-id :_id {conts :Container defins :Definitions exch :Exchange :as m} :Mp}]
109    (let [m (assoc m :mp-id mp-id)]
110      (build-exchange m)
111      (build-meta m)
112      (build-all-container m)
113      (build-all-definitions m)))
114  
115  ;;------------------------------
116  ;; clear mpd doc
117  ;;------------------------------
118  (defn clear-mpd
119    [{mp-id :mp-id}]
120    (stmem/del-vals {:mp-id mp-id :struct :*}))

122  ;;------------------------------
123  ;; build tasks
124  ;;------------------------------
125  (defn build-tasks [tasks]
126    (map (fn [{task-name :TaskName :as task} ]
127           (assoc (stmem/set-val {:task-name task-name
128                                  :value (utils/map->safe-map task)}) :task-name task-name))
129         tasks))
130  
131  ;;------------------------------
132  ;; clear mpd doc
133  ;;------------------------------
134  (defn clear-tasks [] (stmem/del-vals {:task-name :*}))
