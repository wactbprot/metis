001  (ns metis.ltmem.core
002    (:require [metis.config.interface :as c]
003              [com.ashafa.clutch :as couch]
004              [com.brunobonacci.mulog :as mu]
005              [clojure.string :as string]))

007  ;;------------------------------
008  ;; get doc
009  ;;------------------------------
010  (defn get-doc 
011    "Gets a document from the db."
012    ([id]
013     (get-doc c/config id))
014    ([{conn :ltmem-conn} id]
015     (mu/log ::get-doc :message "try to get document" :doc-id id)
016     (try
017       (couch/get-document conn id)
018       (catch Exception e (mu/log ::get-doc :error (.getMessage e) :doc-id id)))))
019    
020  ;;------------------------------
021  ;; put doc
022  ;;------------------------------
023  (defn put-doc
024    "Puts a document to the long term memory."
025    ([doc]
026     (put-doc c/config doc))
027    ([{conn :ltmem-conn} doc]
028     (mu/log ::put-doc :message "try to put document" :doc-id (:_id doc))
029     (try
030       (couch/put-document conn doc)
031       (catch Exception e (mu/log ::put-doc :error (.getMessage e) :doc-id (:_id doc))))))

036  (defn all-tasks

038    ([]

040    ([{conn :ltmem-conn design :ltmem-task-design view :ltmem-task-view}]
041     (mu/log ::all-tasks :message "get tasks from ltm")

043       (mapv :value (couch/get-view conn design view))
044       (catch Exception e (mu/log ::all-tasks :error (.getMessage e))))))

063  (defn exist?
064    "Returns `true` if a document with the `id` exists.
065  
066    TODO: HEAD request not entire doc
067    
068    Example:
069    ```clojure
070    (exist? \"foo-bar\")
071    ;; =>
072    ;; false
073    ```"
074    ([id]
075     (exist? c/config id))
076    ([conf id]
077     (map? (get-doc conf id))))

079  (defn rev-refresh
080    "Refreshs the revision `_rev` of the document if it exist."
081    ([doc]
082     (rev-refresh c/config doc))
083    ([conf doc]
084     (if-let [db-doc (get-doc conf (:_id doc))]
085       (assoc doc :_rev (:_rev db-doc))
086       doc)))
