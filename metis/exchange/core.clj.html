008  (defn path->second-kw
009    "Returns the keyword or nil.
010  
011    ```clojure
012    (key->second-kw \"foo\" )
013    ;; nil
014    (key->second-kw \"foo.bar\" )
015    ;; :bar
016    ```"  
017    [s]
018    (when (string? s)
019      (when-let [x (second (string/split s #"\."))] (keyword x))))
020  
021  (defn path->first-string
022    "Returns the keyword or nil.
023  
024    ```clojure
025    (key->second-kw \"foo\" )
026    ;; nil
027    (key->second-kw \"foo.bar\" )
028    ;; :bar
029    ```"  
030    [s]
031    (when (string? s)
032      (first (string/split s #"\."))))

034  (defn get-val [a p]
035    (or (get a p)
036        (when-let [kw (path->second-kw p)]
037          (kw (get a (path->first-string p))))))

039  (defn ok?
040    "Checks a certain exchange endpoint to evaluate to true"
041    [a p]

060  (defn only-if-not
061    "Runs the task `only-if-not` the exchange path given with `:OnlyIfNot`
062    evaluates to true."
063    [a {p :OnlyIfNot}]
064    (cond
065      (nil? p) true
066      (not (exists? a p)) false
067      (not (ok? a p)) true
068      (ok? a p) false))

070  (defn enclose-map
071    "Encloses the given map `m` with respect to the key `k`.
072  
073    Example:
074    ```clojure
075    (enclose-map {:gg \"ff\"} \"mm.ll\")
076    ;; gives:
077    ;; {\"mm\" {:ll {:gg \"ff\"}}}
078  
079    (enclose-map {:gg \"ff\"} \"mm\")
080    ;; gives:
081    ;; {\"mm\" {:gg \"ff\"}}
082  
083    (enclose-map {\"mm\" \"ff\"} nil)
084    ;; gives:
085    ;; {\"mm\" \"ff\"}
086    ```"
087    [m p]
088    (if p
089      (let [a (path->first-string p)]
090        (if-let [b (path->second-kw p)]
091          {a {b m}}
092          {a m}))
093      m))

095  (defn fit-in [o m]
096    (if o
097      (if (and (map? o) (map? m))
098        (merge o m)
099        m)
100      m))

102  (defn to-vec [a {x :value p :no-idx  :as m}]
103    (into [] (map
104              (fn [[k v]] (assoc m :no-idx (name k) :value (fit-in (get a k) v)))
105              (enclose-map x p))))
