001  (ns metis.exchange.api
002    ^{:author "wactbprot"
003      :doc "Handles the access to the exchange interface."}
004    (:require [metis.exchange.core :as core]
005              [com.brunobonacci.mulog :as mu]
006              [metis.stmem.interface :as stmem]))
007  
008  (defn all [{mp-id :mp-id}]
009    (into {} (map
010              (fn [m] {(:no-idx m) (:value m)})
011              (stmem/get-maps {:mp-id mp-id :struct :exch :no-idx :*}))))
012  
013  (defn to
014    ([m]
015     (to (all m) m))
016    ([a m]
017    (map stmem/set-val (core/to-vec a m))))
018  
019  (defn from
020    "Builds a map by replacing the values of the input map `m`.
021    The replacements are gathered from `a` the complete exchange interface
022    
023    Example:
024    ```clojure
025    (from {\"A\" {:Type \"ref\", :Unit \"Pa\", :Value 100.0},
026           \"B\" \"token\",
027           \"Target_pressure\" {:Selected 1, :Unit \"Pa\"}} {:%check A})
028    ;; =>
029    ;; {:%check {:Type \"ref\" :Unit \"Pa\" :Value 100.0}}
030    ```"
031    ([m]
032     (from (all m) m))
033    ([a m]
034    (when (and (map? a) (map? m))
035      (into {} (map (fn [[k p]] {k (core/get-val a p)}) m)))))
