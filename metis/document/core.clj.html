001  (ns metis.document.core
002    ^{:author "wactbprot"
003      :doc "Handles the documents in which the produced data is stored
004            in.  This may be calibration documents but also measurement
005            docs. The document component needs access to ltmem and
006            stmem."}

015  (defn doc->id [{id :_id}] id)
016  
017  (defn doc->version
018    "Returns the version of the document as an integer value."
019    [{rev :_rev}]
020    (when rev
021      (when-let [v (first (string/split rev  #"-"))] (Integer/parseInt v))))

023  ;;------------------------------
024  ;; extract doc info
025  ;;------------------------------
026  (defn base-info
027    "Returns a map with documents base info."
028    [doc]
029    {:doc-version (doc->version doc) :doc-id (doc->id doc)})
030  
031  (defmulti doc-info
032    "Extracts informations about a document depending on the type."
033    (fn [doc]
034      (first (filter #(not (or (= :_id %) (= :_rev %))) (keys doc)))))
035  
036  (defmethod doc-info :Calibration [doc] (assoc (base-info doc) :doc-type "Calibration"))
037  (defmethod doc-info :Measurement [doc] (assoc (base-info doc) :doc-type "Measurement"))
038  (defmethod doc-info :State       [doc] (assoc (base-info doc) :doc-type "State"))
039  (defmethod doc-info :default     [doc] (assoc (base-info doc) :doc-type "default"))

042  ;; add
043  ;;------------------------------
044  (defn add
045    "Adds a info map to the short term memory."
046    ([m id]
047     (add c/config m id))
048    ([conf {mp-id :mp-id :as m} id]
049     (if-let [doc (ltmem/get-doc conf id)]
050       (do
051         (µ/log ::add :message "doc info added" :m m)
052         (stmem/set-val {:mp-id mp-id :struct :id :doc-id id :value (doc-info doc)}))
053       (do
054         (µ/log ::add :error "document contains no id" :m m)
055         {:error "no document id"}))))
056  
057  ;;------------------------------
058  ;; rm
059  ;;------------------------------
060  (defn rm
061    "Removes the info map from the short term memory."
062    ([m id]
063     (rm c/config m id))
064    ([conf {mp-id :mp-id :as m} id]
065     (µ/log ::rm :message "will rm doc info from stmem" :doc-id id :m m)
066     (stmem/del-val {:mp-id mp-id :struct :id :doc-id id})))
067  
068  ;;------------------------------
069  ;; ids
070  ;;------------------------------
071  (defn ids
072    "Returns the vector of ids added to the short term memory.
073  
074    Example:
075    ```clojure
076    (add {:mp-id \"test\"} \"aaa\")
077    ;; ...
078    (ids \"test\")
079    ;; [aaa]
080    ```"
081    [{mp-id :mp-id :as m}]
082    (mapv (comp :doc-id :value) (stmem/get-maps {:mp-id mp-id :struct :id :doc-id :*})))

088  (defn renew
089    "Renew the id interface with the give ids-vector `v`.
090    TODO: function always returns ok which is not ok."
091    ([m v]
092     (renew c/config m v))
093    ([conf {mp-id :mp-id} v]
094     (when (and (string? mp-id) (vector? v))
095       (let [m {:mp-id mp-id}]
096         (µ/log ::refresh :message "will rm all ids")
097         (doall (mapv #(rm m %) (ids m)))
098         (µ/log ::refresh :message "will add ids provided")
099         (doall (mapv #(add conf m %) v))))
100     {:ok true}))

103  ;;------------------------------
104  ;; store with doc-lock
105  ;;------------------------------
106  (def doc-lock (Object.))
107  
108  (defn execute
109    [conf id results doc-path]
110    (locking doc-lock
111      (µ/log ::execute :message "execute store with lock doc" :doc-id id)
112      (let [d (ltmem/get-doc conf id)
113            d (i/store-results d results doc-path)
114            d (ltmem/put-doc conf d)]
115        {:ok (contains? d :_rev)})))
116  
117  (defn store-results
118    "Stores the `results` vector under the `doc-path` of every document
119    active at the given `mp-id`.
120    
121    Example:
122    ```clojure
123    (def results [{:Type \"cmp-test\" :Unit \"Pa\" :Value 1}
124                 {:Type \"cmp-test2\" :Unit \"Pa\" :Value 2}])
125    
126    (def doc-path  \"Calibration.Measurement.Values.Pressure\")  
127    
128    (store-results {:mp-id \"ref\"} results doc-path)
129    ```"  
130    ([m results doc-path]
131     (store-results c/config m results doc-path))
132    ([conf {mp-id :mp-id :as m} results doc-path]
133     (if (and (string? mp-id) (vector? results) (string? doc-path))
134       (doall
135        (into {} (map #(execute conf % results doc-path) (ids m))))
136       {:error "wrong input params"})))
