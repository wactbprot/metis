014  (defn get-date-object [] (tm/now))
015  (defn get-hour  [d] (tm-f/unparse (tm-f/formatter "HH")   d))
016  (defn get-min   [d] (tm-f/unparse (tm-f/formatter "mm")   d))
017  (defn get-sec   [d] (tm-f/unparse (tm-f/formatter "ss")   d))
018  (defn get-day   [d] (tm-f/unparse (tm-f/formatter "dd")   d))
019  (defn get-month [d] (tm-f/unparse (tm-f/formatter "MM")   d))
020  (defn get-year  [d] (tm-f/unparse (tm-f/formatter "YYYY") d))
021  
022  (defn get-date 
023    ([]
024     (get-date (get-date-object)))
025    ([d]
026     (tm-f/unparse (tm-f/formatters :date) d)))
027  
028  (defn get-time
029    ([]
030     (str (tm-c/to-long (get-date-object))))
031    ([d]
032     (str (tm-c/to-long d))))

037  (defn short-string
038    "Short a `string` `s` to `n` or `45` chars. Returns `nil` is `s` is
039    not a `string`."
040    ([s]
041     (when (string? s) (short-string s 30)))
042    ([s n]
043     (when (string? s) (if (< (count s) n) s (str (subs s 0 n) "...")))))

048   (defn ensure-int
049    "Ensures `i` to be integer. Returns 0 as default."
050    [i]
051    (cond
052      (integer? i) i
053      (string? i) (try (Integer/parseInt i) (catch Exception e 0))))

058  (defn map->safe-map
059    "Replaces all of the `@`-signs (if followed by letters 1)
060    by a `%`-sign  because `:%kw` is a valid keyword but `:@kw` not
061    (or at least problematic).
062  
063    1) There are devices annotating channels by `(@101:105)`.
064    This expressions should remain as they are."
065    ([m]
066     (map->safe-map c/config m))
067    ([{a :at-replace} m]
068    (che/decode (string/replace (che/encode m) #"(@)([a-zA-Z])" (str a "$2")) true)))
069  
070  (defn date-map
071    "Returns a map with replacements
072    of general intrest.
073  
074    ```clojure
075    (globals)
076    ;; {\"%hour\"  \"14\",
077    ;; \"%minute\" \"07\",
078    ;; \"%second\" \"54\",
079    ;; \"%year\"   \"2020\",
080    ;; \"%month\"  \"02\",
081    ;; \"%day\"    \"02\",
082    ;; \"%time\"   \"1580652474824\"}
083    ```"
084    ([]
085     (date-map c/config))
086    ([{a :at-replace}]
087     (let [d (get-date-object)]
088       {(str a "hour") (get-hour d)
089        (str a "minute") (get-min d)
090        (str a "second") (get-sec d)
091        (str a "year") (get-year d)
092        (str a "month") (get-month d)
093        (str a "day") (get-day d)
094        (str a "time") (get-time d)})))
095  
096  (defn apply-to-map-values
097    "Applies function `f` to the values of the map `m`."
098    [f m]
099    (into {} (map (fn [[k v]]
100                    (if (map? v)
101                      [k (apply-to-map-values f v)]
102                      [k (f v)]))
103                  m)))
104  
105  (defn apply-to-map-keys
106    "Applies function `f` to the keys of the map `m`."
107    [f m]
108    (into {} (map (fn [[k v]]
109                    (if (map? v)
110                      [(f k) (apply-to-map-keys f v)]
111                      [(f k) v]))
112                  m)))
