001  (ns metis.stmem.flow-control
002    (:require [metis.stmem.api :as stmem]
003              [com.brunobonacci.mulog :as mu]))
004  
005  ;;------------------------------
006  ;; ctrl
007  ;;------------------------------
008  (defn set-ctrl
009    "Sets the `ctrl` interface to `(:value m)`."
010    [m]
011    (let [kw  (:value m)
012          msg (:message m)]
013      (if (= kw :error )
014        (mu/log ::set-ctrl :error "will set ctrl interface to error" :m m)
015        (mu/log ::set-ctrl :message (str "will set ctrl interface to " kw) :m m))
016      (stmem/set-val (assoc (dissoc m :par-idx :seq-idx) :func :ctrl :value (name kw)))))
017  
018  ;;------------------------------
019  ;; state
020  ;;------------------------------
021  (defn set-states 
022    "Sets all states (the state interface) to almost ready."
023    [m]
024    (let [kw  (:value m)
025          msg (:message m)]
026      (mu/log ::set-states :message (or msg  "will set all states") :m m)
027      (stmem/set-vals (assoc m :func :state :seq-idx :* :par-idx :* :value (name kw)))))
028  
029  (defn set-state
030    "Sets the state to `(:value m)`."
031    [m] 
032    (let [kw  (:value m)
033          msg (:message m)]
034      (if (= kw :error )
035        (mu/log ::set-state :error (or msg "will set stat interface to error") :m m)
036        (mu/log ::set-state :message (or msg (str "will set state interface to " kw)) :m m))
037      (stmem/set-val (assoc m :func :state :value (name kw)))))
