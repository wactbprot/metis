001  (ns metis.stmem.core
002    (:require [metis.config.interface :as c]
003              [taoensso.carmine :as car :refer (wcar)]
004              [cheshire.core :as che]
005              [com.brunobonacci.mulog :as mu]
006              [clojure.string :as string]))
007  
008  ;;------------------------------
009  ;; stmem set
010  ;;------------------------------
011  (defn set-val
012    "Sets the value `v` for the key `k`."
013    ([k v]
014     (set-val c/config k v))  
015    ([{conn :stmem-conn} k v]
016     (wcar conn (car/set k v))))
017  
018  (defn set-vals
019    "Sets the given `val` for all keys `ks` with the delay `mtp`."
020    [ks v]
021    (run! (fn [k] (set-val k v)) ks))
022  
023  ;;------------------------------
024  ;; get keys
025  ;;------------------------------
026  (defn pat->keys
027    "Get all keys matching  the given pattern `pat`."
028    ([pat]
029     (pat->keys c/config pat))
030    ([{conn :stmem-conn} pat]
031     (sort (wcar conn (car/keys pat)))))
032   
033  ;;------------------------------
034  ;; del
035  ;;------------------------------
036  (defn del-val
037    "Deletes the key `k`."
038    ([k]
039     (del-val c/config k))
040    ([{conn :stmem-conn} k]
041     (wcar conn (car/del k))))
042  
043  (defn del-vals
044    "Deletes all given keys (`ks`)."
045    [ks]
046    (run! del-val ks))
047  
048  ;;------------------------------
049  ;; get value(s)
050  ;;------------------------------
051  (defn get-val
052    "Returns the value for the given key (`k`) and cast it to a clojure
053    type."
054    ([k]
055     (get-val c/config k))
056    ([{conn :stmem-conn} k]
057     (wcar conn (car/get k))))
058  
059  (defn get-vals
060    "Returns a vector of the `vals` behind the keys `ks`."
061    [ks]
062    (mapv get-val ks))
063  
064  (defn filter-keys-where-val
065    "Returns a list of all keys belonging to the pattern `pat` where the
066    value is equal to`x`.
067    
068    Example:
069    ```clojure
070    (filter-keys-where-val \"ref@definitions@*@class\" \"wait\")
071    ;; (\"ref@definitions@0@class\"
072    ;; \"ref@definitions@2@class\"
073    ;; \"ref@definitions@1@class\")
074    ```"
075    [pat x]
076    (filter (fn [k] (= x (get-val k))) (pat->keys pat)))
