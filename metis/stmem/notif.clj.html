016  ;; register key utils

018  (defn reg-pat
019    ([m]
020     (reg-pat c/config m))
021    ([{t :stmem-trans s :stmem-notif-sep :as config} m]
022     (let [mp-id (:mp-id m)
023           struct ((or (:struct m) :*) t)
024           no-idx (:no-idx m)
025           no-idx (if (number? no-idx) (trans/lpad config no-idx) (:* t))
026           func ((or (:func m) :*) t)]
027       (str mp-id s struct s no-idx s func))))
028     
029  (defn subs-pat
030    "Generates subscribe patterns."
031    ([m]
032     (subs-pat c/config m))
033    ([config m]
034     (let [{t :stmem-trans s :stmem-notif-sep db :stmem-db} config]
035       (str "__keyspace" s db (:* t) "__:" (reg-pat config m) (:* t)))))
036  
037  (defn reg-key
038    ([m]
039     (reg-key c/config m))
040    ([{t :stmem-trans s :stmem-notif-sep :as config} m]
041     (when (and (map? m) (seq m))
042       (str (reg-pat config m) s (trans/lpad config (or (:level m)  0))))))

047  (defn wrap-skip-psubs
048    "Ensures that f is only executed on `pmessage`s. Skips the subscribe
049    event calls."
050    [f]
051    (fn [v] (when (= "pmessage" (first v)) (f v))))
052  
053  (defn wrap-msg->key
054    [f]
055    (fn [[_ _ s _]]
056      (f (second (string/split s #":")))))
057  
058  (defn wrap-key->map
059    ([f]
060     (wrap-key->map c/config f))
061    ([config f]
062     (fn [k] (f (trans/key->map k)))))
063    
064  (defn wrap-assoc-value
065    ([f]
066     (wrap-assoc-value c/config f))
067    ([config f]
068     (fn [m]
069       (let [m (assoc m :value (api/get-val m))]
070         (mu/log ::wrap-assoc-value :message  "listener callback, assoc value" :m m)
071         (f m)))))

074  ;; generate listener
075  ;;------------------------------
076  (defn gen-listener
077    "Returns a listener for published keyspace **notif**ications. Don't forget
078    to [[close-listener]]
079  
080     Example:
081    ```clojure
082    ;; generate and close
083    (close-listener! (gen-listener {:mp-id \"ref\"} msg->key))
084    ```"
085    ([m f]
086     (gen-listener c/config m f))
087    ([{{conn :spec} :stmem-conn :as config} m f]
088     (let [pat (subs-pat config m)
089           f (->> f
090                  wrap-assoc-value
091                  wrap-key->map
092                  wrap-msg->key
093                  wrap-skip-psubs)]
094       (mu/log ::gen-listener :message (str "will subscribe to: " pat))
095       (car/with-new-pubsub-listener conn {pat f} (car/psubscribe pat))))) 
096  
097  (defn registered? [k] (contains? @listeners k))
098  
099  (defn register
100    "Generates a listener for the function `f`. Registers it at the
101    `listeners` atom."
102    ([m f]
103     (register c/config m f))
104    ([config m f]
105     (let [reg-key (reg-key config m)]
106       (if-not (registered? reg-key)
107         (do
108          (swap! listeners assoc reg-key (gen-listener m f))
109          (Thread/sleep (:stmem-reg-relax config))
110          {:ok true})
111         {:ok true :warn "already registered"}))))
112  
113  ;;------------------------------
114  ;; close listeners, deregister 
115  ;;------------------------------
116  (defn close-listener
117    "Closes the given listener generated by [[gen-listener]].
118  
119    Example:
120    ```clojure
121    ;; generate
122    (def l (gen-listener {:mp-id \"ref\"} msg->map))
123    ;; close 
124    (close-listener! l)
125    ```"
126    [l]
127    (car/close-listener l))
128  
129  (defn clean-register
130    "Closes and `de-registers` **all** `listeners` belonging to `mp-id` ."
131    [m]
132    (map (fn [[k v]]
133           (when (string/starts-with? k (:mp-id m))
134              (mu/log ::clean-register :message "will clean" :key k)
135             (close-listener v)
136             (swap! listeners dissoc k)))
137         @listeners))
138  
139  (defn de-register
140    "De-registers the listener with the key `mp-id` in the `listeners`
141    atom."
142    [m]
143    (let [k (reg-key m)]
144      (mu/log ::de-register :message "will de-register" :key k)
145      (when (registered? k)
146        (close-listener (@listeners k))
147        {:ok (map? (swap! listeners dissoc k))})))
