001  (ns metis.stmem.api
002    (:require [metis.stmem.core :as core]
003              [metis.config.interface :as c]
004              [cheshire.core :as che]
005              [clojure.string :as string]
006              [metis.stmem.trans :as trans]))
007  
008  (defn get-val
009    ([m]
010     (get-val c/config m))
011    ([config m]
012     (che/parse-string (core/get-val (trans/map->key m)) true)))
013  
014  (defn get-map
015    ([m]
016     (get-val c/config m))
017    ([config m]
018     (assoc m :value (get-val config m))))
019  
020  (defn get-maps
021    ([m]
022     (get-maps c/config m))
023    ([config m]
024     (mapv (fn [k]
025             (merge
026              (assoc m :value (che/parse-string (core/get-val k) true))
027              (trans/key->map k)))
028           (core/pat->keys (trans/map->key m)))))
029  
030  (defn set-val
031    ([m]
032     (set-val c/config m))
033    ([{relax :stmem-mod-relax} m]
034     (if-let [val (:value m)]
035       (do 
036         (core/set-val (trans/map->key m) (che/generate-string val))
037         (Thread/sleep relax)
038         {:ok true})
039       {:error "no value given"})))
040  
041  (defn set-vals
042    ([m]
043     (set-vals c/config m))
044    ([{relax :stmem-mod-relax} m]
045     (if-let [val (:value m)]
046       (do 
047         (core/set-vals (core/pat->keys (trans/map->key m)) (che/generate-string val))
048         (Thread/sleep relax)
049         {:ok true})
050       {:error "no value given"})))
051  
052  (defn del-val
053    ([m]
054     (del-val c/config m))
055    ([{relax :stmem-mod-relax} m]
056     (core/del-val (trans/map->key m))
057     (Thread/sleep relax)
058     {:ok true}))
059  
060  (defn del-vals
061    ([m]
062     (del-vals c/config m))
063    ([config m]
064     (core/del-vals (core/pat->keys (trans/map->key m)))
065     {:ok true}))
