011  (defn pad-ok?
012    "Checks if the padding of `i` is ok. `\"*\"` serves pattern matching."
013    ([idx]
014     (pad-ok? c/config idx))
015    ([config idx]
016     (cond
017       (and (string? idx)
018            (= (count idx)
019               (:stmem-key-pad-length config))) true
020       (= idx "*")                         true 
021       :else                               false)))
022  
023  (defn lpad
024    "Left pad the given number if it is not a string."
025    ([idx]
026     (lpad c/config idx))
027    ([config idx]
028     (if (pad-ok? idx)
029       idx
030       (when idx (format (str "%0" (:stmem-key-pad-length config) "d") (utils/ensure-int idx))))))

032  (defn map->task-key
033    [{trans :stmem-trans s :stmem-key-sep} m]
034    (let [task-name (:task-name m)]
035      (str (:tasks trans) s (if (keyword? task-name) (task-name trans) task-name))))
036  
037  (defn map->struct-part
038    [{trans :stmem-trans s :stmem-key-sep} m]
039    (when  (:mp-id m)
040      (when-let [p (:struct m)]
041        (str (:mp-id m) s (if (keyword? p) (p trans) p)))))
042  
043  (defn map->no-idx-part
044    [{trans :stmem-trans s :stmem-key-sep :as config} m]
045    (when-let[no-idx (:no-idx m)]
046      (str s (if (= :* no-idx) (:* trans) (lpad config (:no-idx m))))))
047  
048  (defn map->exch-part
049    [{s :stmem-key-sep :as config} m]
050    (when-let [p (:exch m)] (str s p)))
051

053    [{trans :stmem-trans s :stmem-key-sep :as config} m]
054    (when-let [p (:doc-id m)] (str s (if (keyword? p) (p trans) p))))

063  
064  (defn map->seq-par-idx-part
065    [{trans :stmem-trans s :stmem-key-sep :as config} m]
066    (let [sdx (:seq-idx m)
067          pdx (:par-idx m)]
068      (when sdx
069      (str s (if (number? sdx) (lpad config sdx) (:* trans))
070           (when pdx
071             (str s (if (number? pdx) (lpad config pdx) (:* trans))))))))
072  
073  (defn map->def-key
074    [config m]
075    (str
076     (map->struct-part config m)
077     (str ;this str only groups the next parts
078      (map->no-idx-part config m)
079      (map->exch-part config m)
080      ( map->doc-id-part config m)
081      (map->metapath-part config m)) 
082     (map->func-part config m)
083     (map->seq-par-idx-part config m)))
084  
085  (defn map->key
086    ([m]
087     (map->key c/config m))
088    ([config m]
089     (when (and (map? m) (seq m)) 
090       (if (:task-name m)
091           (map->task-key config m)
092           (map->def-key config m)))))

098  (defn care-no-idx [config v]
099    (when-let [s (get v 2)]
100      (if (re-matches #"[0-9]*" s) (utils/ensure-int s) s)))

102  (defn care-struct [{retrans :stmem-retrans} v] (get retrans  (get v 1)))
103  
104  (defn key->map
105    ([k]
106     (key->map c/config k))
107    ([{re-sep :re-sep retrans :stmem-retrans :as config} k]
108     (when k
109       (when-let [v (string/split k re-sep)]
110         {:mp-id (get v 0)
111          :struct (care-struct config v)
112          :no-idx (care-no-idx config v)
113          :func (get retrans (get v 3))
114          :seq-idx (utils/ensure-int (get v 4))
115          :par-idx (utils/ensure-int (get v 5))}))))
